#define LEFT_SENSOR 34
#define FRONT_SENSOR 35
#define RIGHT_SENSOR 32

void setup() {
Serial.begin(115200);

pinMode(LEFT_SENSOR, INPUT);
pinMode(FRONT_SENSOR, INPUT);
pinMode(RIGHT_SENSOR, INPUT);
}

void loop() {

int L = digitalRead(LEFT_SENSOR);
int F = digitalRead(FRONT_SENSOR);
int R = digitalRead(RIGHT_SENSOR);

Serial.print("Left: ");
Serial.print(L);

Serial.print("  Front: ");
Serial.print(F);

Serial.print("  Right: ");
Serial.println(R);

delay(300);
}
